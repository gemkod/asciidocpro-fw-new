////
////
===============================================================================
Préparation des lignes de métadonnée
===============================================================================
////
////
-------------------------------------------------------------------------------
Préparation de la première ligne de métadonnée
-------------------------------------------------------------------------------
////
////
Identifiant du support
////
:_tpl_id:
ifdef::_meta_id[]
ifeval::["{_meta_id}" != ""]
:_tpl_id: {_meta_id}
:_tpl_id: Id : {_tpl_id}
//end eval _meta_id
endif::[]
endif::_meta_id[]
////
Numéro de version
////
:_tpl_version_number:
ifdef::_meta_version_number[]
ifeval::["{_meta_version_number}" != ""]
:_tpl_version_number: Version {_meta_version_number}
ifeval::["{_tpl_id}" != ""]
:_tpl_version_number:  | {_tpl_version_number}
//end eval _tpl_id
endif::[]
//end eval _meta_version_number
endif::[]
endif::_meta_version_number[]
////
Date de version
////
:_tpl_version_date:
ifdef::_meta_version_date[]
ifeval::["{_meta_version_date}" != ""]
:_tpl_version_date: {_meta_version_date}
ifeval::["{_tpl_version_number}" != ""]
:_tpl_version_date:  du {_meta_version_date}
//end eval _tpl_version_number
endif::[]
ifeval::["{_tpl_version_number}" == ""]
ifeval::["{_tpl_id}" != ""]
:_tpl_version_date:  | Version du {_meta_version_date}
//end eval _tpl_id
endif::[]
ifeval::["{_tpl_id}" == ""]
:_tpl_version_date: Version du {_meta_version_date}
//end eval _tpl_id
endif::[]
//end eval _tpl_version_number
endif::[]
//end eval _meta_version_date
endif::[]
endif::_meta_version_date[]
////
Date de révision
////
:_tpl_revision_date:
ifdef::_meta_revision_date[]
ifeval::["{_meta_revision_date}" != ""]
:_tpl_revision_date: Dernière révision : {_meta_revision_date}
ifeval::["{_tpl_version_date}" != ""]
:_tpl_revision_date:  | {_tpl_revision_date}
//end eval _tpl_version_date
endif::[]
ifeval::["{_tpl_version_date}" == ""]
ifeval::["{_tpl_version_number}" != ""]
:_tpl_revision_date:  | {_tpl_revision_date}
//end eval _tpl_version_number
endif::[]
//end eval _tpl_version_date
endif::[]
ifeval::["{_tpl_version_date}" == ""]
ifeval::["{_tpl_version_number}" == ""]
ifeval::["{_tpl_id}" != ""]
:_tpl_revision_date:  | {_tpl_revision_date}
//end eval _tpl_id
endif::[]
//end eval _tpl_version_number
endif::[]
//end eval _tpl_version_date
endif::[]
//end eval _meta_revision_date
endif::[]
endif::_meta_revision_date[]
////
-------------------------------------------------------------------------------
Préparation de la seconde ligne de métadonnée
-------------------------------------------------------------------------------
////
// Durée de réalisation
:_tpl_completion_time:
ifdef::_meta_completion_time[]
ifeval::["{_meta_completion_time}" != ""]
:_tpl_completion_time: Durée de réalisation : {_meta_completion_time}
//end eval _meta_completion_time
endif::[]
endif::_meta_completion_time[]
// Durée de correction
:_tpl_correction_time:
ifdef::_meta_correction_time[]
ifeval::["{_meta_correction_time}" != ""]
:_tpl_correction_time: Durée de correction : {_meta_correction_time}
ifeval::["{_tpl_completion_time}" != ""]
:_tpl_correction_time:  | {_tpl_correction_time}
//end eval _tpl_completion_time
endif::[]
//end eval _meta_correction_time
endif::[]
endif::_meta_correction_time[]
// Durée totale
:_tpl_total_time:
ifdef::_meta_total_time[]
ifeval::["{_meta_total_time}" != ""]
:_tpl_total_time: Durée totale {_meta_total_time}
ifeval::["{_tpl_correction_time}" != ""]
:_tpl_total_time:  | {_tpl_total_time}
//end eval _tpl_correction_time
endif::[]
ifeval::["{_tpl_correction_time}" == ""]
ifeval::["{_tpl_completion_time}" != ""]
:_tpl_total_time:  | {_tpl_total_time}
//end eval _tpl_completion_time
endif::[]
//end eval _tpl_correction_time
endif::[]
//end eval _meta_total_time
endif::[]
endif::_meta_total_time[]
////
-------------------------------------------------------------------------------
Préparation de la ligne 3
-------------------------------------------------------------------------------
////
:_tpl_author:
ifdef::_meta_author[]
:_tpl_author: {_meta_author}
ifeval::["{_tpl_author}" != ""]
:_tpl_author: Auteur : {_tpl_author}
//end eval _tpl_author
endif::[]
:_need_of_separator:
endif::_meta_author[]
////
===============================================================================
Gestion de l'affichage des lignes
===============================================================================
////
////
-------------------------------------------------------------------------------
Affectation du contenu à chaque ligne
-------------------------------------------------------------------------------
////
:_line_1: {_tpl_id}{_tpl_version_number}{_tpl_version_date}{_tpl_revision_date}
:_line_2: {_tpl_completion_time}{_tpl_correction_time}{_tpl_total_time}
:_line_3: {_tpl_author}
////
-------------------------------------------------------------------------------
Affichage des métadonnées en fonction de l'état vide ou non de chaque ligne
-------------------------------------------------------------------------------
////
//Une ligne vide doit être affichée pour séparer les métadonnées des attributs de document.

// Si seule la ligne 1 est présente
ifeval::["{_line_1}" != ""]
ifeval::["{_line_2}" == ""]
ifeval::["{_line_3}" == ""]
[.metadata]#_{_line_1}_#
//end eval _line_3
endif::[]
//end eval _line_2
endif::[]
//end eval _line_1
endif::[]
//
// Si seule la ligne 2 est présente
ifeval::["{_line_1}" == ""]
ifeval::["{_line_2}" != ""]
ifeval::["{_line_3}" == ""]
[.metadata]#_{_line_2}_#
//end eval _line_3
endif::[]
//end eval _line_2
endif::[]
//end eval _line_1
endif::[]
//
// Si seule la ligne 3 est présente
ifeval::["{_line_1}" == ""]
ifeval::["{_line_2}" == ""]
ifeval::["{_line_3}" != ""]
[.metadata]#_{_line_3}_#
//end eval _line_3
endif::[]
//end eval _line_2
endif::[]
//end eval _line_1
endif::[]
//
// Si les lignes 1 et 2 sont présentes
ifeval::["{_line_1}" != ""]
ifeval::["{_line_2}" != ""]
ifeval::["{_line_3}" == ""]
[.metadata]#_{_line_1}_# +
[.metadata]#_{_line_2}_#
//end eval _line_3
endif::[]
//end eval _line_2
endif::[]
//end eval _line_1
endif::[]
//
// Si les lignes 1 et 3 sont présentes
ifeval::["{_line_1}" != ""]
ifeval::["{_line_2}" == ""]
ifeval::["{_line_3}" != ""]
[.metadata]#_{_line_1}_# +
[.metadata]#_{_line_3}_#
//end eval _line_3
endif::[]
//end eval _line_2
endif::[]
//end eval _line_1
endif::[]
//
// Si les lignes 2 et 3 sont présentes
ifeval::["{_line_1}" == ""]
ifeval::["{_line_2}" != ""]
ifeval::["{_line_3}" != ""]
[.metadata]#_{_line_2}_# +
[.metadata]#_{_line_3}_#
//end eval _line_3
endif::[]
//end eval _line_2
endif::[]
//end eval _line_1
endif::[]
//
// Si les trois lignes sont présentes
ifeval::["{_line_1}" != ""]
ifeval::["{_line_2}" != ""]
ifeval::["{_line_3}" != ""]
[.metadata]#_{_line_1}_# +
[.metadata]#_{_line_2}_# +
[.metadata]#_{_line_3}_#
//end eval _line_3
endif::[]
//end eval _line_2
endif::[]
//end eval _line_1
endif::[]
//Ajout du chemin

ifeval::["{_meta_content_path}" != ""]

_Indication de l'emplacement du dossier dans le projet AsciidocPro :_ +
`{_meta_content_path}`
//end eval _meta_container_path
endif::[]
endif::_meta_content_path[]
